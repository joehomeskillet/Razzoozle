47/// Configuration for Socket.IO & Engine.IO
48#[derive(Debug, Clone)]
49pub struct SocketIoConfig {
50    /// The inner Engine.IO config
51    pub engine_config: EngineIoConfig,
52
53    /// The amount of time the server will wait for an acknowledgement from the client before closing the connection.
54    ///
55    /// Defaults to 5 seconds.

57
58    /// The amount of time before disconnecting a client that has not successfully joined a namespace.
59    ///
60    /// Defaults to 45 seconds.

81/// A builder to create a [`SocketIo`] instance.

278/// The [`SocketIo`] instance can be cheaply cloned and moved around everywhere in your program.
279/// It can be used as the main handle to access the whole socket.io context.
280///
281/// You can also use it as an extractor for all your [`handlers`](crate::handler).
282#[derive(Debug)]
283pub struct SocketIo<A: Adapter = LocalAdapter>(Arc<Client<A>>);

289        SocketIoBuilder::new()

292    /// Creates a new [`SocketIoService`] and a [`SocketIo`] instance with a default config.
293    /// This service will be a _standalone_ service that return a 404 error for every non-socket.io request.
294    /// It can be used as a [`Service`](tower_service::Service) (see hyper example)
295    #[inline(always)]
296    pub fn new_svc() -> (SocketIoService<NotFoundService>, SocketIo) {

302    #[inline(always)]
303    pub fn new_inner_svc<S: Clone>(svc: S) -> (SocketIoService<S>, SocketIo) {
304        Self::builder().build_with_inner_svc(svc)
305    }

307    /// Builds a [`SocketIoLayer`] and a [`SocketIo`] instance with a default config.
308    /// It can be used as a tower [`Layer`](tower_layer::Layer) (see axum example)
309    #[inline(always)]

312    }

322    /// Registers a [`ConnectHandler`] for the given namespace
323    ///
324    /// * See the [`connect`](crate::handler::connect) module doc for more details on connect handler.
325    /// * See the [`extract`](crate::extract) module doc for more details on available extractors.
326    ///
327    /// # Examples
328    /// #### Simple example with a sync closure:
329    /// ```
330    /// # use socketioxide::{SocketIo, extract::*};
331    /// # use serde::{Serialize, Deserialize};
332    /// #[derive(Debug, Serialize, Deserialize)]
333    /// struct MyData {
334    ///     name: String,
335    ///     age: u8,
336    /// }
337    ///
338    /// let (_, io) = SocketIo::new_svc();
339    /// io.ns("/", |socket: SocketRef| {
340    ///     // Register a handler for the "test" event and extract the data as a `MyData` struct
341    ///     // With the Data extractor, the handler is called only if the data can be deserialized as a `MyData` struct
342    ///     // If you want to manage errors yourself you can use the TryData extractor
343    ///     socket.on("test", |socket: SocketRef, Data::<MyData>(data)| {
344    ///         println!("Received a test message {:?}", data);
345    ///         socket.emit("test-test", &MyData { name: "Test".to_string(), age: 8 }).ok(); // Emit a message to the client
346    ///     });
347    /// });
348    ///
349    /// ```
350    ///
351    /// #### Example with a closure and an acknowledgement + binary data:
352    /// ```
353    /// # use socketioxide::{SocketIo, extract::*};
354    /// # use serde_json::Value;
355    /// # use serde::{Serialize, Deserialize};
356    /// #[derive(Debug, Serialize, Deserialize)]
357    /// struct MyData {
358    ///     name: String,
359    ///     age: u8,
360    /// }
361    ///
362    /// let (_, io) = SocketIo::new_svc();
363    /// io.ns("/", |socket: SocketRef| {
364    ///     // Register an async handler for the "test" event and extract the data as a `MyData` struct
365    ///     // Extract the binary payload as a `Vec<Bytes>` with the Bin extractor.
366    ///     // It should be the last extractor because it consumes the request
367    ///     socket.on("test", |socket: SocketRef, Data::<MyData>(data), ack: AckSender| async move {
368    ///         println!("Received a test message {:?}", data);
369    ///         tokio::time::sleep(std::time::Duration::from_secs(1)).await;
370    ///         ack.send(&data).ok(); // The data received is sent back to the client through the ack
371    ///         socket.emit("test-test", &MyData { name: "Test".to_string(), age: 8 }).ok(); // Emit a message to the client
372    ///     });
373    /// });
374    /// ```
375    /// #### Simple example with a closure:
376    /// ```
377    /// # use socketioxide::{SocketIo, extract::{SocketRef, Data}};
378    /// # use serde::{Serialize, Deserialize};
379    /// #[derive(Debug, Deserialize)]
380    /// struct MyAuthData {
381    ///     token: String,
382    /// }
383    /// #[derive(Debug, Serialize, Deserialize)]
384    /// struct MyData {
385    ///     name: String,
386    ///     age: u8,
387    /// }
388    ///
389    /// let (_, io) = SocketIo::new_svc();
390    /// io.ns("/", |socket: SocketRef, Data(auth): Data<MyAuthData>| {
391    ///     if auth.token.is_empty() {
392    ///         println!("Invalid token, disconnecting");
393    ///         socket.disconnect().ok();
394    ///         return;
395    ///     }
396    ///     socket.on("test", |socket: SocketRef, Data::<MyData>(data)| async move {
397    ///         println!("Received a test message {:?}", data);
398    ///         socket.emit("test-test", &MyData { name: "Test".to_string(), age: 8 }).ok(); // Emit a message to the client
399    ///     });
400    /// });
401    ///
402    /// ```
403    #[inline]
404    pub fn ns<C, T>(&self, path: impl Into<Cow<'static, str>>, callback: C)
405    where
406        C: ConnectHandler<A, T>,
407        T: Send + Sync + 'static,
408    {
409        self.0.add_ns(path.into(), callback);
410    }

412    /// Registers a [`ConnectHandler`] for the given dynamic namespace.
413    /// You can specify dynamic parts in the path by using the `{name}` syntax.
414    /// Note that any static namespace will take precedence over a dynamic one.
415    ///
416    ///
417    /// For more info about namespace routing, see the [matchit] router documentation.
418    ///
419    /// The dynamic namespace will create a child namespace for any path that matches the given pattern with the given handler.
420    ///
421    /// * See the [`connect`](crate::handler::connect) module doc for more details on connect handler.
422    /// * See the [`extract`](crate::extract) module doc for more details on available extractors.
423    ///
424    /// ## Errors
425    /// If the pattern is invalid, a [`NsInsertError`](crate::NsInsertError) will be returned.
426    ///
427    /// ## Example
428    /// ```
429    /// # use socketioxide::{SocketIo, extract::SocketRef};
430    /// let (_, io) = SocketIo::new_svc();
431    /// io.dyn_ns("/client/{client_id}", |socket: SocketRef| {
432    ///     println!("Socket connected on dynamic namespace with namespace path: {}", socket.ns());
433    /// }).unwrap();
434    ///
435    /// ```
436    /// ```
437    /// # use socketioxide::{SocketIo, extract::SocketRef};
438    /// let (_, io) = SocketIo::new_svc();
439    /// io.dyn_ns("/client/{*remaining_path}", |socket: SocketRef| {
440    ///     println!("Socket connected on dynamic namespace with namespace path: {}", socket.ns());
441    /// }).unwrap();
442    ///
443    /// ```
444    #[inline]
445    pub fn dyn_ns<C, T>(
446        &self,
447        path: impl Into<String>,
448        callback: C,
449    ) -> Result<(), crate::NsInsertError>
450    where
451        C: ConnectHandler<A, T>,
452        T: Send + Sync + 'static,
453    {
454        self.0.add_dyn_ns(path.into(), callback)
455    }

628    /// [`emit_with_ack()`]: #method.emit_with_ack
629    ///
630    /// # Example
631    /// ```
632    /// # use socketioxide::{SocketIo, extract::SocketRef};
633    /// # use futures_util::stream::StreamExt;
634    /// # use std::time::Duration;
635    /// # use serde_json::Value;
636    /// let (_, io) = SocketIo::new_svc();
637    /// io.ns("/", |socket: SocketRef| {
638    ///     println!("Socket connected on / namespace with id: {}", socket.id);
639    /// });
640    ///
641    /// // Later in your code you can emit a test message on the root namespace in the room1 and room3 rooms,
642    /// // except for the room2 and wait for 5 seconds for an acknowledgement
643    /// io.to("room1")
644    ///   .to("room3")
645    ///   .except("room2")
646    ///   .timeout(Duration::from_secs(5))
647    ///   .emit_with_ack::<_, Value>("message-back", "I expect an ack in 5s!")
648    ///   .unwrap()
649    ///   .for_each(|(sid, ack)| async move {
650    ///      match ack {
651    ///          Ok(ack) => println!("Ack received, socket {} {:?}", sid, ack),
652    ///          Err(err) => println!("Ack error, socket {} {:?}", sid, err),
653    ///      }
654    ///   });
655    #[inline]
656    pub fn timeout(&self, timeout: Duration) -> BroadcastOperators<A> {
657        self.get_default_op().timeout(timeout)
658    }

660    /// Emits a message to all sockets selected with the previous operators.
661    ///
662    /// If you provide tuple-like data (tuple, arrays), it will be considered as multiple arguments.
663    /// Therefore if you want to send an array as the _first_ argument of the payload,
664    /// you need to wrap it in an array or a tuple. [`Vec`] will be always considered as a single argument though.
665    ///
666    /// ## Emitting binary data
667    /// To emit binary data, you must use a data type that implements [`Serialize`] as binary data.
668    /// Currently if you use `Vec<u8>` it will be considered as a number sequence and not binary data.
669    /// To counter that you must either use a special type like [`Bytes`] or use the [`serde_bytes`] crate.
670    /// If you want to emit generic data that may contains binary, use [`rmpv::Value`] rather
671    /// than [`serde_json::Value`] otherwise the binary data will also be serialized as a number sequence.
672    ///
673    /// ## Errors
674    /// * When encoding the data a [`BroadcastError::Serialize`] may be returned.
675    /// * If the underlying engine.io connection is closed for a given socket a [`BroadcastError::Socket(SocketError::Closed)`]
676    ///   will be returned.
677    /// * If the packet buffer is full for a given socket, a [`BroadcastError::Socket(SocketError::InternalChannelFull)`]
678    ///   will be retured. See [`SocketIoBuilder::max_buffer_size`] option for more infos on internal buffer config
679    ///
680    /// > **Note**: If a error is returned because of a specific socket, the message will still be sent to all other sockets.
681    ///
682    /// [`SocketIoBuilder::max_buffer_size`]: crate::SocketIoBuilder#method.max_buffer_size
683    /// [`BroadcastError::Socket(SocketError::Closed)`]: crate::SocketError::Closed
684    /// [`BroadcastError::Socket(SocketError::InternalChannelFull)`]: crate::SocketError::InternalChannelFull
685    /// [`Bytes`]: bytes::Bytes
686    /// [`serde_bytes`]: https://docs.rs/serde_bytes
687    /// [`rmpv::Value`]: https://docs.rs/rmpv
688    /// [`serde_json::Value`]: https://docs.rs/serde_json/latest/serde_json/value
689    ///
690    /// ## Panics
691    /// If the **default namespace "/" is not found** this fn will panic!
692    ///
693    /// ## Example
694    /// ```
695    /// # use socketioxide::{SocketIo, extract::SocketRef};
696    /// # use serde_json::Value;
697    /// let (_, io) = SocketIo::new_svc();
698    /// io.ns("/", |socket: SocketRef| {
699    ///     println!("Socket connected on / namespace with id: {}", socket.id);
700    /// });
701    ///
702    /// // Later in your code you can emit a test message on the root namespace in the room1 and room3 rooms,
703    /// // except for the room2
704    /// io.to("room1")
705    ///   .to("room3")
706    ///   .except("room2")
707    ///   .emit("Hello World!", &());
708    /// ```
709    ///
710    /// ## Binary Example with the `bytes` crate
711    /// ```
712    /// # use socketioxide::{SocketIo, extract::*};
713    /// # use serde_json::Value;
714    /// # use std::sync::Arc;
715    /// # use bytes::Bytes;
716    /// let (_, io) = SocketIo::new_svc();
717    /// io.ns("/", |socket: SocketRef| {
718    ///     socket.on("test", |socket: SocketRef, Data::<(String, Bytes, Bytes)>(data)| async move {
719    ///         // Emit a test message to the client
720    ///         socket.emit("test", &data).ok();
721    ///
722    ///         // Emit a test message with multiple arguments to the client
723    ///         socket.emit("test", &("world", "hello", Bytes::from_static(&[1, 2, 3, 4]))).ok();
724    ///
725    ///         // Emit a test message with an array as the first argument
726    ///         let arr = [1, 2, 3, 4];
727    ///         socket.emit("test", &[arr]).ok();
728    ///     });
729    /// });
730    /// ```
731    #[inline]
732    pub fn emit<T: ?Sized + Serialize>(
733        &self,
734        event: impl AsRef<str>,
735        data: &T,
736    ) -> Result<(), BroadcastError> {
737        self.get_default_op().emit(event, data)
738    }

740    /// Emits a message to all sockets selected with the previous operators and
741    /// waits for the acknowledgement(s).
742    ///
743    /// See [`emit()`](#method.emit) for more details on how to emit data.
744    ///
745    /// To get acknowledgements, an [`AckStream`] is returned.
746    /// It can be used in two ways:
747    /// * As a [`Stream`]: It will yield all the ack responses with their corresponding socket id
748    ///   received from the client. It can useful when broadcasting to multiple sockets and therefore expecting
749    ///   more than one acknowledgement. If you want to get the socket from this id, use [`io::get_socket()`].
750    /// * As a [`Future`]: It will yield the first ack response received from the client.
751    ///   Useful when expecting only one acknowledgement.
752    ///
753    /// If the packet encoding failed an [`EncodeError`] is **immediately** returned.
754    ///
755    /// If the socket is full or if it has been closed before receiving the acknowledgement,
756    /// an [`AckError::Socket`] will be yielded.
757    ///
758    /// If the client didn't respond before the timeout, the [`AckStream`] will yield
759    /// an [`AckError::Timeout`]. If the data sent by the client is not deserializable as `V`,
760    /// an [`AckError::Decode`] will be yielded.
761    ///
762    /// [`timeout()`]: #method.timeout
763    /// [`Stream`]: futures_core::stream::Stream
764    /// [`Future`]: futures_core::future::Future
765    /// [`AckError::Decode`]: crate::AckError::Decode
766    /// [`AckError::Timeout`]: crate::AckError::Timeout
767    /// [`AckError::Socket`]: crate::AckError::Socket
768    /// [`AckError::Socket(SocketError::Closed)`]: crate::SocketError::Closed
769    /// [`EncodeError`]: crate::EncodeError
770    /// [`io::get_socket()`]: crate::SocketIo#method.get_socket
771    ///
772    /// # Panics
773    /// If the **default namespace "/" is not found** this fn will panic!
774    ///
775    /// # Example
776    /// ```
777    /// # use socketioxide::{SocketIo, extract::*};
778    /// # use serde_json::Value;
779    /// # use futures_util::stream::StreamExt;
780    /// let (_, io) = SocketIo::new_svc();
781    /// io.ns("/", |socket: SocketRef| {
782    ///     socket.on("test", |socket: SocketRef, Data::<Value>(data)| async move {
783    ///         // Emit a test message in the room1 and room3 rooms,
784    ///         // except for the room2 room with the binary payload received
785    ///         let ack_stream = socket.to("room1")
786    ///             .to("room3")
787    ///             .except("room2")
788    ///             .emit_with_ack::<_, String>("message-back", &data)
789    ///             .unwrap();
790    ///
791    ///         ack_stream.for_each(|(sid, ack)| async move {
792    ///             match ack {
793    ///                 Ok(ack) => println!("Ack received, socket {} {:?}", sid, ack),
794    ///                 Err(err) => println!("Ack error, socket {} {:?}", sid, err),
795    ///             }
796    ///         }).await;
797    ///     });
798    /// });
799    #[inline]
800    pub fn emit_with_ack<T: ?Sized + Serialize, V>(
801        &self,
802        event: impl AsRef<str>,
803        data: &T,
804    ) -> Result<AckStream<V>, parser::EncodeError> {
805        self.get_default_op().emit_with_ack(event, data)
806    }
