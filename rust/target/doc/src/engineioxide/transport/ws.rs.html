36/// Create a response for websocket upgrade
37fn ws_response<B>(ws_key: &HeaderValue) -> Result<Response<ResponseBody<B>>, http::Error> {
38    let derived = derive_accept_key(ws_key.as_bytes());
39    let sec = derived.parse::<HeaderValue>().unwrap();
40    Response::builder()
41        .status(StatusCode::SWITCHING_PROTOCOLS)
42        .header(http::header::UPGRADE, HeaderValue::from_static("websocket"))
43        .header(
44            http::header::CONNECTION,
45            HeaderValue::from_static("Upgrade"),
46        )
47        .header(http::header::SEC_WEBSOCKET_ACCEPT, sec)
48        .body(ResponseBody::empty_response())
49}

51/// Upgrade a websocket request to create a websocket connection.
52///
53/// If a sid is provided in the query it means that is is upgraded from an existing HTTP polling request.
54/// In this case the http polling request is closed and the SID is kept for the websocket
55pub fn new_req<R: Send + 'static, B, H: EngineIoHandler>(
56    engine: Arc<EngineIo<H>>,
57    protocol: ProtocolVersion,
58    sid: Option<Sid>,
59    req: Request<R>,
60) -> Result<Response<ResponseBody<B>>, Error> {
61    let (parts, body) = req.into_parts();
62    let req = Request::from_parts(parts.clone(), body);
63
64    let ws_key = parts
65        .headers
66        .get("Sec-WebSocket-Key")
67        .ok_or(Error::HttpErrorResponse(StatusCode::BAD_REQUEST))?
68        .clone();
69
70    tokio::spawn(async move {
71        let conn = hyper::upgrade::on(req)
72            .await
73            .map(hyper_util::rt::TokioIo::new);
74        let res = match conn {
75            Ok(conn) => on_init(engine, conn, protocol, sid, parts).await,
76            Err(_e) => {
77                #[cfg(feature = "tracing")]
78                tracing::debug!("ws upgrade error: {}", _e);
79                return;
80            }
81        };
82
83        match res {
84            Ok(_) => {
85                #[cfg(feature = "tracing")]
86                tracing::debug!("ws closed")
87            }
88            Err(_e) => {
89                #[cfg(feature = "tracing")]
90                tracing::debug!("ws closed with error: {:?}", _e)
91            }
92        }
93    });
94
95    Ok(ws_response(&ws_key)?)
96}

98/// Handle a websocket connection upgrade
99///
100/// Sends an open packet if it is not an upgrade from a polling request
101///
102/// Read packets from the websocket and handle them, it will block until the connection is closed
103async fn on_init<H: EngineIoHandler, S>(
104    engine: Arc<EngineIo<H>>,
105    conn: S,
106    protocol: ProtocolVersion,
107    sid: Option<Sid>,
108    req_data: Parts,
109) -> Result<(), Error>
110where
111    S: AsyncRead + AsyncWrite + Unpin + Send + 'static,
112{
113    let ws_init = move || WebSocketStream::from_raw_socket(conn, Role::Server, None);
114    let (socket, ws) = if let Some(sid) = sid {
115        match engine.get_socket(sid) {
116            None => return Err(Error::UnknownSessionID(sid)),
117            Some(socket) if socket.is_ws() => return Err(Error::Upgrade),
118            Some(socket) => {
119                let mut ws = ws_init().await;
120                upgrade_handshake::<H, S>(&socket, &mut ws).await?;
121                (socket, ws)
122            }
123        }
124    } else {
125        let socket = engine.create_session(
126            protocol,
127            TransportType::Websocket,
128            req_data,
129            #[cfg(feature = "v3")]
130            false,
131        );
132        #[cfg(feature = "tracing")]
133        tracing::debug!("[sid={}] new websocket connection", socket.id);
134        let mut ws = ws_init().await;
135        init_handshake(socket.id, &mut ws, &engine.config).await?;
136        socket
137            .clone()
138            .spawn_heartbeat(engine.config.ping_interval, engine.config.ping_timeout);
139        (socket, ws)
140    };
141    let (tx, rx) = ws.split();
142    let rx_handle = forward_to_socket::<H, S>(socket.clone(), tx);
143
144    if let Err(ref e) = forward_to_handler(&engine, rx, &socket).await {
145        #[cfg(feature = "tracing")]
146        tracing::debug!("[sid={}] error when handling packet: {:?}", socket.id, e);
147        if let Some(reason) = e.into() {
148            engine.close_session(socket.id, reason);
149        }
150    } else {
151        engine.close_session(socket.id, DisconnectReason::TransportClose);
152    }
153    rx_handle.abort();
154    Ok(())
155}

157/// Forwards all packets received from a websocket to a EngineIo [`Socket`]
158async fn forward_to_handler<H: EngineIoHandler, S>(
159    engine: &Arc<EngineIo<H>>,
160    mut rx: SplitStream<WebSocketStream<S>>,
161    socket: &Arc<Socket<H::Data>>,
162) -> Result<(), Error>
163where
164    S: AsyncRead + AsyncWrite + Unpin + Send + 'static,
165{
166    while let Some(msg) = rx.try_next().await? {
167        match msg {
168            Message::Text(msg) => match Packet::try_from(msg)? {
169                Packet::Close => {
170                    #[cfg(feature = "tracing")]
171                    tracing::debug!("[sid={}] closing session", socket.id);
172                    engine.close_session(socket.id, DisconnectReason::TransportClose);
173                    break;
174                }
175                Packet::Pong | Packet::Ping => socket
176                    .heartbeat_tx
177                    .try_send(())
178                    .map_err(|_| Error::HeartbeatTimeout),
179                Packet::Message(msg) => {
180                    engine.handler.on_message(msg, socket.clone());
181                    Ok(())
182                }
183                p => return Err(Error::BadPacket(p)),
184            },
185            Message::Binary(mut data) => {
186                if socket.protocol == ProtocolVersion::V3 && !data.is_empty() {
187                    // The first byte is the message type, which we don't need.
188                    let _ = data.remove(0);
189                }
190                engine.handler.on_binary(data.into(), socket.clone());
191                Ok(())
192            }
193            Message::Close(_) => break,
194            _ => {
195                #[cfg(feature = "tracing")]
196                tracing::debug!("[sid={}] unexpected ws message", socket.id);
197                Ok(())
198            }
199        }?
200    }
201    Ok(())
202}

204/// Forwards all packets waiting to be sent to the websocket
205///
206/// The websocket stream is flushed only when the internal channel is drained
207fn forward_to_socket<H: EngineIoHandler, S>(
208    socket: Arc<Socket<H::Data>>,
209    mut tx: SplitSink<WebSocketStream<S>, Message>,
210) -> JoinHandle<()>
211where
212    S: AsyncRead + AsyncWrite + Unpin + Send + 'static,
213{
214    // Pipe between websocket and internal socket channel
215    tokio::spawn(async move {
216        let mut internal_rx = socket.internal_rx.try_lock().unwrap();
217
218        // map a packet to a websocket message
219        // It is declared as a macro rather than a closure to avoid ownership issues
220        macro_rules! map_fn {
221            ($item:ident) => {
222                let res = match $item {
223                    Packet::Binary(bin) | Packet::BinaryV3(bin) => {
224                        let mut bin: Vec<u8> = bin.into();
225                        if socket.protocol == ProtocolVersion::V3 {
226                            // v3 protocol requires packet type as the first byte
227                            bin.insert(0, 0x04);
228                        }
229                        tx.feed(Message::Binary(bin)).await
230                    }
231                    Packet::Close => {
232                        tx.send(Message::Close(None)).await.ok();
233                        internal_rx.close();
234                        break;
235                    },
236                    // A Noop Packet maybe sent by the server to upgrade from a polling connection
237                    // In the case that the packet was not poll in time it will remain in the buffer and therefore
238                    // it should be discarded here
239                    Packet::Noop => Ok(()),
240                    _ => {
241                        let packet: String = $item.try_into().unwrap();
242                        tx.feed(Message::Text(packet)).await
243                    }
244                };
245                if let Err(_e) = res {
246                    #[cfg(feature = "tracing")]
247                    tracing::debug!("[sid={}] error sending packet: {}", socket.id, _e);
248                }
249            };
250        }
251
252        while let Some(items) = internal_rx.recv().await {
253            for item in items {
254                map_fn!(item);
255            }
256            // For every available packet we continue to send until the channel is drained
257            while let Ok(items) = internal_rx.try_recv() {
258                for item in items {
259                    map_fn!(item);
260                }
261            }
262
263            tx.flush().await.ok();
264        }
265    })
266}
267/// Send a Engine.IO [`OpenPacket`] to initiate a websocket connection
268async fn init_handshake<S>(
269    sid: Sid,
270    ws: &mut WebSocketStream<S>,
271    config: &EngineIoConfig,
272) -> Result<(), Error>
273where
274    S: AsyncRead + AsyncWrite + Unpin + Send + 'static,
275{
276    let packet = Packet::Open(OpenPacket::new(TransportType::Websocket, sid, config));
277    ws.send(Message::Text(packet.try_into()?)).await?;
278    Ok(())
279}

281/// Upgrade a session from a polling request to a websocket request.
282///
283/// Before upgrading the session the server should send a NOOP packet to any pending polling request.
284///
285/// ## Handshake :
286/// ```text
287/// CLIENT                                                 SERVER
288///│                                                      │
289///│   GET /engine.io/?EIO=4&transport=websocket&sid=...  │
290///│ ───────────────────────────────────────────────────► │
291///│  ◄─────────────────────────────────────────────────┘ │
292///│            HTTP 101 (WebSocket handshake)            │
293///│                                                      │
294///│            -----  WebSocket frames -----             │
295///│  ─────────────────────────────────────────────────►  │
296///│                         2probe                       │ (ping packet)
297///│  ◄─────────────────────────────────────────────────  │
298///│                         3probe                       │ (pong packet)
299///│  ─────────────────────────────────────────────────►  │
300///│                         5                            │ (upgrade packet)
301///│                                                      │
302///│            -----  WebSocket frames -----             │
303/// ```
304#[cfg_attr(feature = "tracing", tracing::instrument(skip(socket, ws), fields(sid = socket.id.to_string())))]
305async fn upgrade_handshake<H: EngineIoHandler, S>(
306    socket: &Arc<Socket<H::Data>>,
307    ws: &mut WebSocketStream<S>,
308) -> Result<(), Error>
309where
310    S: AsyncRead + AsyncWrite + Unpin + Send + 'static,
311{
312    #[cfg(feature = "tracing")]
313    tracing::debug!("websocket connection upgrade");
314
315    // Fetch the next packet from the ws stream, it should be a PingUpgrade packet
316    let msg = match ws.next().await {
317        Some(Ok(Message::Text(d))) => d,
318        _ => Err(Error::Upgrade)?,
319    };
320    match Packet::try_from(msg)? {
321        Packet::PingUpgrade => {
322            // Respond with a PongUpgrade packet
323            ws.send(Message::Text(Packet::PongUpgrade.try_into()?))
324                .await?;
325        }
326        p => Err(Error::BadPacket(p))?,
327    };
328
329    // send a NOOP packet to any pending polling request so it closes gracefully
330    socket.send(Packet::Noop)?;
331
332    // Fetch the next packet from the ws stream, it should be an Upgrade packet
333    let msg = match ws.next().await {
334        Some(Ok(Message::Text(d))) => d,
335        Some(Ok(Message::Close(_))) => {
336            #[cfg(feature = "tracing")]
337            tracing::debug!("ws stream closed before upgrade");
338            Err(Error::Upgrade)?
339        }
340        _ => {
341            #[cfg(feature = "tracing")]
342            tracing::debug!("unexpected ws message before upgrade");
343            Err(Error::Upgrade)?
344        }
345    };
346    match Packet::try_from(msg)? {
347        Packet::Upgrade => {
348            #[cfg(feature = "tracing")]
349            tracing::debug!("ws upgraded successful")
350        }
351        p => Err(Error::BadPacket(p))?,
352    };
353
354    // wait for any polling connection to finish by waiting for the socket to be unlocked
355    let _ = socket.internal_rx.lock().await;
356    socket.upgrade_to_websocket();
357    Ok(())
